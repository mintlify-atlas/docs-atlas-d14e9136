## Overview

Bulk Processing allows you to analyze sentiment for multiple text entries simultaneously by uploading a CSV file. This feature is ideal for processing large volumes of customer feedback, reviews, or comments.

## API Endpoint

```
POST /sentiment/upload-csv
```

### Authentication

All requests require a Bearer token in the Authorization header:

```bash
Authorization: Bearer {your_token}
```

## File Requirements

<Info>
The CSV file must be under **3MB** in size. Files exceeding this limit will be rejected.
</Info>

### CSV Format

Your CSV file should contain text entries to be analyzed. The specific column structure depends on your API configuration.

## Request Format

The endpoint accepts multipart form data with the CSV file:

```typescript
const formData = new FormData();
formData.append('file', selectedFile);
```

### Headers

```json
{
  "Authorization": "Bearer {your_token}"
}
```

<Note>
Do NOT set `Content-Type` header manually. The browser will automatically set it to `multipart/form-data` with the appropriate boundary.
</Note>

## Response Format

The API returns an array of analyzed results:

```typescript
interface AnalisisResultado {
  texto: string;
  prevision: 'POSITIVO' | 'NEGATIVO';
  probabilidad: number;
}
```

### Response Example

```json
[
  {
    "texto": "Great product, highly recommend!",
    "prevision": "POSITIVO",
    "probabilidad": 0.9456
  },
  {
    "texto": "Terrible service, very disappointed.",
    "prevision": "NEGATIVO",
    "probabilidad": 0.8823
  }
]
```

```python Python
import requests

def upload_csv(file_path: str, token: str):
    url = f"{API_URL}/sentiment/upload-csv"
    
    # Check file size (3MB limit)
    import os
    file_size = os.path.getsize(file_path)
    max_size = 3 * 1024 * 1024
    
    if file_size > max_size:
        raise ValueError(f"File too large: {file_size / (1024*1024):.2f} MB")
    
    headers = {"Authorization": f"Bearer {token}"}
    
    with open(file_path, 'rb') as f:
        files = {'file': f}
        response = requests.post(url, files=files, headers=headers)
    
    response.raise_for_status()
    return response.json()
```

## Error Handling

The endpoint returns specific HTTP status codes for different error conditions:

| Status Code | Description | Cause |
|-------------|-------------|-------|
| 401 | Unauthorized | File exceeds 3MB limit OR invalid token |
| 413 | Payload Too Large | File size exceeds server limit |
| 403 | Forbidden | Insufficient permissions |
| 500 | Internal Server Error | Server processing error |

### Comprehensive Error Handling

```typescript
try {
  const response = await fetch(
    `${process.env.NEXT_PUBLIC_API_URL}/sentiment/upload-csv`,
    {
      method: "POST",
      headers: { "Authorization": `Bearer ${token}` },
      body: formData,
    }
  );

  if (!response.ok) {
    switch (response.status) {
      case 401:
        throw new Error(
          "El archivo supera el límite de 3MB. " +
          "Intenta con un archivo más pequeño."
        );
      case 413:
        throw new Error(
          "El servidor indica que el archivo es demasiado grande."
        );
      case 403:
        throw new Error(
          "No tienes permisos para realizar esta acción."
        );
      case 500:
        throw new Error(
          "Error interno del servidor. Inténtalo más tarde."
        );
      default:
        throw new Error(
          `Ocurrió un problema inesperado (Error ${response.status})`
        );
    }
  }

  const data = await response.json();
  return data;
  
} catch (err) {
  if (err.message === "Failed to fetch") {
    console.error("No se pudo conectar con el servidor.");
  } else {
    console.error(err.message);
  }
  throw err;
}
```

```tsx
const [selectedFile, setSelectedFile] = useState<File | null>(null);
const [uploading, setUploading] = useState(false);
const [uploadStatus, setUploadStatus] = useState<'success' | 'error' | null>(null);

const handleUpload = async (e: React.FormEvent) => {
  e.preventDefault();
 
  if (!token) {
    setMessage("Sesión expirada. Por favor, inicia sesión nuevamente.");
    setUploadStatus('error');
    return;
  }

  if (!selectedFile) {
    setMessage("Por favor, selecciona un archivo primero.");

    return;
  }

  // File size validation

  if (selectedFile.size > MAX_SIZE_MB) {

    return;
  }

  setUploading(true);
  setUploadStatus(null);
  setMessage('');

  const formData = new FormData();
  formData.append('file', selectedFile);

  try {
    const response = await fetch(
      `${process.env.NEXT_PUBLIC_API_URL}/sentiment/upload-csv`,
      {
        method: "POST",
        headers: { "Authorization": `Bearer ${token}` },
        body: formData,
      }
    );

    if (!response.ok) {
      switch (response.status) {
        case 401:
          throw new Error(
            "El archivo supera el límite de 3MB. " +
            "Intenta con un archivo más pequeño."
          );
        case 413:
          throw new Error("El servidor indica que el archivo es demasiado grande.");
        case 403:
          throw new Error("No tienes permisos para realizar esta acción.");
        case 500:
          throw new Error("Error interno del servidor. Inténtalo más tarde.");
        default:
          throw new Error(`Ocurrió un problema inesperado (Error ${response.status})`);
      }
    }

    const data = await response.json();
    setResultados(data);
    setMessage('¡Excelente! El archivo se procesó correctamente.');
    setUploadStatus('success');
    setSelectedFile(null); 
    
  } catch (err: any) {
    if (err.message === "Failed to fetch") {
      setMessage("No se pudo conectar con el servidor. Verifica tu conexión.");
    } else {
      setMessage(err.message);
    }
    setUploadStatus('error');
  } finally {
    setUploading(false);
  }
};
```
