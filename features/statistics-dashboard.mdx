## Overview

The Statistics Dashboard provides aggregate sentiment analysis data for previously analyzed comments stored in the database. You can retrieve the most recent comments with their sentiment classifications and view summary statistics.

## API Endpoint

```
GET /stats
```

### Authentication

All requests require a Bearer token in the Authorization header:

```bash
Authorization: Bearer {your_token}
```

## Query Parameters

| Parameter | Type | Required | Description |
|-----------|------|----------|-------------|
| `size` | number | Yes | Number of recent comments to retrieve |

### Example Request

```bash
GET /stats?size=50
```

## Response Format

The API returns a `SentimentApiResponse` object:

```typescript
interface SentimentApiResponse {
  total_en_pagina: number;      // Total comments in current page
  positivos: string;             // Percentage of positive comments
  negativos: string;             // Percentage of negative comments
  content: SentimentResult[];    // Array of individual results
  totalElements: number;         // Total elements available
  totalPages: number;            // Total pages available
  number: number;                // Current page number
}

interface SentimentResult {
  id: number;
  comentario: string;
  prevision: 'POSITIVO' | 'NEGATIVO';
  probabilidad: number;
  fecharegistro: string;
}
```

## Field Descriptions

<Steps>
  <Step title="total_en_pagina">
    The total number of comments returned in the current response
  </Step>
  <Step title="positivos">
    Percentage of positive sentiment comments as a string (e.g., "64.5")
  </Step>
  <Step title="negativos">
    Percentage of negative sentiment comments as a string (e.g., "35.5")
  </Step>
  <Step title="content">
    Array of individual comment analysis results with timestamps
  </Step>
  <Step title="totalElements">
    Total number of analyzed comments available in the database
  </Step>
  <Step title="totalPages">
    Number of pages available for pagination
  </Step>
  <Step title="number">
    Current page number (0-indexed)
  </Step>
</Steps>

## Full Implementation Example

```tsx
interface SentimentResult {
  id: number;
  comentario: string;
  prevision: 'POSITIVO' | 'NEGATIVO';
  probabilidad: number;
  fecharegistro: string;
}

interface SentimentApiResponse {
  total_en_pagina: number;
  positivos: string;
  negativos: string;
  content: SentimentResult[];
  totalElements: number;
  totalPages: number;
  number: number;
}

const [comment, setComment] = useState('');
const [dataResponse, setDataResponse] = useState<SentimentApiResponse | null>(null);
const [resultado, setResultado] = useState<SentimentResult[]>([]);
const [loading, setLoading] = useState(false);
const [message, setMessage] = useState('');

const handleAnalyze = async (e: React.FormEvent) => {
  e.preventDefault();
  setLoading(true);

  const numComent = parseInt(comment);

  try {
    const response = await fetch(
      `${process.env.NEXT_PUBLIC_API_URL}/stats?size=${numComent}`,
      {
        method: "GET",
        headers: {
          "Content-Type": "application/json",
          "Authorization": `Bearer ${token}`
        },
      }
    );

    const data = await response.json();
    setDataResponse(data);
    setResultado(data?.content);
  } catch (err: any) {
    if (err.message === "Failed to fetch") {
      setMessage("No se pudo conectar con el servidor. Verifica tu conexión.");
    } else {
      setMessage(err.message);
    }
  } finally {
    setLoading(false);
  }
};
```

## Displaying Statistics

### Summary Cards

Display aggregate statistics in visual cards:

```tsx
{/* Total Comments */}
<div className="stat-card">
  <span>Total Comentarios</span>
  <p className="stat-value">{dataResponse?.total_en_pagina ?? 0}</p>
</div>

{/* Positive Percentage */}
<div className="stat-card positive">
  <span>Positivos</span>
  <p className="stat-value">
    {dataResponse?.positivos 
      ? parseFloat(dataResponse.positivos).toFixed(1) 
      : 0}%
  </p>
</div>

{/* Negative Percentage */}
<div className="stat-card negative">
  <span>Negativos</span>
  <p className="stat-value">
    {dataResponse?.negativos 
      ? parseFloat(dataResponse.negativos).toFixed(1) 
      : 0}%
  </p>
</div>
```

### Results Table

Display individual comment results:

```tsx
<table>
  <thead>
    <tr>
      <th>Comentario</th>
      <th>Prevision</th>
      <th>Probabilidad</th>
    </tr>
  </thead>
  <tbody>
    {resultado?.map((res, index) => (
      <tr key={index}>
        <td>{res.comentario}</td>
        <td>
          <span className={res.prevision === 'POSITIVO' ? 'positive' : 'negative'}>
            {res.prevision}
          </span>
        </td>
        <td>
          <div className="probability">
            <span>{(res.probabilidad * 100).toFixed(1)}%</span>
            <div className="progress-bar">
              <div 
                className={res.prevision === 'POSITIVO' ? 'positive-bar' : 'negative-bar'}
                style={{ width: `${res.probabilidad * 100}%` }}
              />
            </div>
          </div>
        </td>
      </tr>
    ))}
  </tbody>
</table>
```

## Pagination

<Info>
The response includes pagination metadata (`totalPages`, `totalElements`, `number`) for implementing paginated views.
</Info>

To implement pagination, you would typically add a `page` parameter to the request:

```typescript
const fetchPage = async (page: number, size: number, token: string) => {
  const response = await fetch(
    `${process.env.NEXT_PUBLIC_API_URL}/stats?page=${page}&size=${size}`,
    {
      method: "GET",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${token}`
      },
    }
  );
  return await response.json();
};
```

## Error Handling

```typescript
try {
  const response = await fetch(
    `${process.env.NEXT_PUBLIC_API_URL}/stats?size=${size}`,
    {
      method: "GET",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${token}`
      },
    }
  );

  if (!response.ok) {
    switch (response.status) {
      case 401:
        throw new Error("Sesión expirada. Por favor, inicia sesión.");
      case 403:
        throw new Error("No tienes permisos para ver estas estadísticas.");
      case 500:
        throw new Error("Error del servidor. Intenta más tarde.");
      default:
        throw new Error(`Error inesperado: ${response.status}`);
    }
  }

  const data = await response.json();
  return data;
} catch (error) {
  if (error.message === "Failed to fetch") {
    console.error("No se pudo conectar con el servidor.");
  }
  throw error;
}
```
